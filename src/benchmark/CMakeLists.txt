set(BENCHMARKS
  BitInterleaving128bitBenchmark
  BitInterleaving64bitBenchmark
  CoordinateConversionBenchmark
  EventCreationBenchmark
  MDBoxBenchmark
  SortBenchmark
  SortEventBenchmark
)

foreach(BENCHMARK ${BENCHMARKS})

    NAME ${BENCHMARK}
